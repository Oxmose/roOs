77 OS_RETURN_E console_set_selected_driver(const kernel_console_driver_t* driver)
78 {
79 #ifdef ARCH_64_BITS
80  KERNEL_TRACE_EVENT(EVENT_KERNEL_CONSOLE_SET_DRIVER_START, 2,
81  (uintptr_t)driver & 0xFFFFFFFF,
82  (uintptr_t)driver >> 32);
83 #else
84  KERNEL_TRACE_EVENT(EVENT_KERNEL_CONSOLE_SET_DRIVER_START, 2,
85  (uintptr_t)driver,
86  0);
87 #endif
88 
89  if(driver == NULL ||
90  driver->clear_screen == NULL ||
91  driver->put_string == NULL ||
92  driver->put_char == NULL)
93  {
94  KERNEL_TRACE_EVENT(EVENT_KERNEL_CONSOLE_SET_DRIVER_END, 1,
95  OS_ERR_NULL_POINTER);
96  return OS_ERR_NULL_POINTER;
97  }
98 
99  console_driver = *driver;
100 
101  KERNEL_TRACE_EVENT(EVENT_KERNEL_CONSOLE_SET_DRIVER_END, 1,
102  OS_NO_ERR);
103  return OS_NO_ERR;
104 }
