309 static fdt_property_t* parseProperty(uint32_t* pOffset,
310  fdt_node_t* pNode)
311 {
312  fdt_property_t* pProperty;
313  const char* pName;
314  size_t length;
315  const uint32_t* pInitProperty;
316 
317  /* Check if start property */
318  if(FDTTOCPU32(fdtDesc.pStructs[*pOffset]) != FDT_PROP)
319  {
320  return NULL;
321  }
322  ++(*pOffset);
323 
324  /* Allocate new property */
325  pProperty = kmalloc(sizeof(fdt_property_t));
326  if(pProperty == NULL)
327  {
328  PANIC(OS_ERR_NO_MORE_MEMORY,
329  MODULE_NAME,
330  "Failed to allocate new property",
331  TRUE);
332  }
333  memset(pProperty, 0, sizeof(fdt_property_t));
334 
335 
336  pInitProperty = (uint32_t*)&fdtDesc.pStructs[*pOffset];
337  *pOffset += 2;
338 
339  /* Get the length and name */
340  pProperty->length = FDTTOCPU32(pInitProperty[0]);
341  pName = fdtDesc.pStrings + FDTTOCPU32(pInitProperty[1]);
342  length = strlen(pName);
343 
344  pProperty->pName = kmalloc(length + 1);
345  if(pProperty->pName == NULL)
346  {
347  PANIC(OS_ERR_NO_MORE_MEMORY,
348  MODULE_NAME,
349  "Failed to allocate new string",
350  TRUE);
351  }
352  memcpy(pProperty->pName, pName, length);
353  pProperty->pName[length] = 0;
354 
355  /* Copy the property cells */
356  if(pProperty->length != 0)
357  {
358  pProperty->pCells = kmalloc(pProperty->length);
359  if(pProperty->pCells == NULL)
360  {
361  PANIC(OS_ERR_NO_MORE_MEMORY,
362  MODULE_NAME,
363  "Failed to allocate new property cells",
364  TRUE);
365  }
366  memcpy(pProperty->pCells, fdtDesc.pStructs + *pOffset, pProperty->length);
367  }
368  else
369  {
370  pProperty->pCells = NULL;
371  }
372 
373  KERNEL_DEBUG(DTB_DEBUG_ENABLED,
374  MODULE_NAME,
375  "Read property %s of length %u",
376  pProperty->pName,
377  pProperty->length);
378 
379  *pOffset += ALIGN(pProperty->length, FDT_CELL_SIZE) / FDT_CELL_SIZE;
380 
381  applyPropertyAction(pNode, pProperty);
382 
383  return pProperty;
384 }

386 static fdt_node_t* parseNode(uint32_t* pOffset,
387  const uint8_t kAddrCells,
388  const uint8_t kSizeCells)
389 {
390  fdt_node_t* pNode;
391  fdt_node_t* pChildNode;
392  fdt_property_t* pProperty;
393  const char* pInitName;
394  size_t length;
395  uint32_t cursor;
396 
397  /* Check if start node */
398  if(FDTTOCPU32(fdtDesc.pStructs[*pOffset]) != FDT_BEGIN_NODE)
399  {
400  return NULL;
401  }
402 
403  ++(*pOffset);
404 
405  /* Allocate new node */
406  pNode = kmalloc(sizeof(fdt_node_t));
407  if(pNode == NULL)
408  {
409  PANIC(OS_ERR_NO_MORE_MEMORY,
410  MODULE_NAME,
411  "Failed to allocate new node",
412  TRUE);
413  }
414  memset(pNode, 0, sizeof(fdt_node_t));
415 
416  pNode->addrCells = kAddrCells;
417  pNode->sizeCells = kSizeCells;
418 
419  /* Get name and copy */
420  pInitName = (char*)&fdtDesc.pStructs[*pOffset];
421  length = strlen(pInitName);
422 
423  pNode->pName = kmalloc(length + 1);
424  if(pNode->pName == NULL)
425  {
426  PANIC(OS_ERR_NO_MORE_MEMORY,
427  MODULE_NAME,
428  "Failed to allocate new string",
429  TRUE);
430  }
431  memcpy(pNode->pName, pInitName, length);
432  pNode->pName[length] = 0;
433 
434  KERNEL_DEBUG(DTB_DEBUG_ENABLED, MODULE_NAME, "Read node %s", pNode->pName);
435 
436  /* Update offset */
437  *pOffset += ALIGN(length + 1, FDT_CELL_SIZE) / FDT_CELL_SIZE;
438 
439  /* Walk the node until we reache the number of structs */
440  while(fdtDesc.nbStructs > *pOffset)
441  {
442  cursor = FDTTOCPU32(fdtDesc.pStructs[*pOffset]);
443  if(cursor == FDT_BEGIN_NODE)
444  {
445  /* New Child */
446  pChildNode = parseNode(pOffset, pNode->addrCells, pNode->sizeCells);
447  if(pChildNode != NULL)
448  {
449  if(pNode->pFirstChildNode == NULL)
450  {
451  pChildNode->pNextNode = NULL;
452  pNode->pFirstChildNode = pChildNode;
453  }
454  else
455  {
456  linkNode(pNode->pFirstChildNode, pChildNode);
457  }
458  }
459  }
460  else if(cursor == FDT_PROP)
461  {
462  pProperty = parseProperty(pOffset, pNode);
463  if(pProperty != NULL)
464  {
465  if(pNode->pProps == NULL)
466  {
467  pProperty->pNextProp = NULL;
468  pNode->pProps = pProperty;
469  }
470  else
471  {
472  linkProperty(pNode->pProps, pProperty);
473  }
474  }
475  }
476  else
477  {
478  ++(*pOffset);
479  if(cursor == FDT_END_NODE)
480  {
481  return pNode;
482  }
483  }
484  }
485 
486  KERNEL_ERROR("Unexpected end of node\n");
487 
488  kfree(pNode->pName);
489  kfree(pNode);
490 
491  return NULL;
492 }

495 {
496  uint32_t i;
497 
498  const fdt_header_t* pHeader;
499  fdt_node_t* pNode;
500 
501  KERNEL_DEBUG(DTB_DEBUG_ENABLED,

504  kStartAddr);
505 
506  pHeader = (fdt_header_t*)kStartAddr;
507 
508  /* Check magic */
509  if(FDTTOCPU32(pHeader->magic) != FDT_MAGIC_NUMBER)
510  {
511  PANIC(OS_ERR_INCORRECT_VALUE,
512  MODULE_NAME,
513  "Invalid FDT magic",
514  TRUE);
515  }
516 
517  memset(&fdtDesc, 0, sizeof(fdt_descriptor_t));
518 
519  /* Get the structs and strings addresses */
520  fdtDesc.pStructs = (uint32_t*)(kStartAddr + FDTTOCPU32(pHeader->offStructs));
521  fdtDesc.pStrings = (char*)(kStartAddr + FDTTOCPU32(pHeader->offStrings));
522 
523  fdtDesc.nbStructs = FDTTOCPU32(pHeader->sizeStructs) / sizeof(uint32_t);
524 
525  /* Now start parsing the first level */
526  for(i = 0; i < fdtDesc.nbStructs; ++i)
527  {
528  /* Get the node and add to root */
529  pNode = parseNode(&i, INIT_ADDR_CELLS, INIT_SIZE_CELLS);
530  if(pNode != NULL)
531  {
532  /* Link node */
533  if(fdtDesc.pFirstNode == NULL)
534  {
535  pNode->pNextNode = NULL;
536  fdtDesc.pFirstNode = pNode;
537  }
538  else
539  {
540  linkNode(fdtDesc.pFirstNode, pNode);
541  }
542  }
543  }
