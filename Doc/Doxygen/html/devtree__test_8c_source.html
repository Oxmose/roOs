80 void walkNodes(const fdt_node_t* pkNode, const uint8_t kLevel)
81 {
82  uint8_t i;
83  const fdt_property_t* pProp;
84 
85  if(pkNode == NULL)
86  {
87  return;
88  }
89 
90  for(i = 0; i < kLevel; ++i)
91  {
92  kernel_printf(" ");
93  }
94  kernel_printf("-> %s\n", pkNode->pName);
95  pProp = fdtGetFirstProp(pkNode);
96  while(pProp != NULL)
97  {
98  for(i = 0; i < kLevel; ++i)
99  {
100  kernel_printf(" ");
101  }
102 
103  kernel_printf(" | %s\n", pProp->pName);
104  pProp = fdtGetNextProp(pProp);
105  }
106  walkNodes(fdtGetChild(pkNode), kLevel + 1);
107  walkNodes(fdtGetNextNode(pkNode), kLevel);
108 }
