84  * longword to upper half of long, i.e., produce the upper longword of

106 /*
107  * __qdivrem(u, v, rem) returns u/v and, optionally, sets *rem to u%v.
108  *
109  * We do this in base 2-sup-HALF_BITS, so that all intermediate products
110  * fit within uint32_t. As a consequence, the maximum length dividend and
111  * divisor are 4 `digits' in this base (they are shorter if they have
112  * leading zeros).
113  */
114 uint64_t
115 __qdivrem(uint64_t uq, uint64_t vq, uint64_t *arq)
116 {
117  union uu tmp;
118  digit *u, *v, *q;
119  digit v1, v2;
120  uint32_t qhat, rhat, t;
121  int m, n, d, j, i;
122  digit uspace[5], vspace[5], qspace[5];
123 
124  /*
125  * Take care of special cases: divide by zero, and u < v.
126  */
127  if (vq == 0) {
128  /* divide by zero. */
129  //static volatile const unsigned int zero = 0;
130 
131  //tmp.ul[H] = tmp.ul[L] = 1 / zero;
132  if (arq)
133  *arq = uq;
134  tmp.q = 0;
135  return (tmp.q);
136  }
137  if (uq < vq) {
138  if (arq)
139  *arq = uq;
140  return (0);
141  }
142  u = &uspace[0];
143  v = &vspace[0];
144  q = &qspace[0];
145 
146  /*
147  * Break dividend and divisor into digits in base B, then
148  * count leading zeros to determine m and n. When done, we
149  * will have:
150  * u = (u[1]u[2]...u[m+n]) sub B
151  * v = (v[1]v[2]...v[n]) sub B
152  * v[1] != 0
153  * 1 < n <= 4 (if n = 1, we use a different division algorithm)
154  * m >= 0 (otherwise u < v, which we already checked)
155  * m + n = 4
156  * and thus
157  * m = 4 - n <= 2
158  */
159  tmp.uq = uq;
160  u[0] = 0;
161  u[1] = (digit)HHALF(tmp.ul[H]);
162  u[2] = (digit)LHALF(tmp.ul[H]);
163  u[3] = (digit)HHALF(tmp.ul[L]);
164  u[4] = (digit)LHALF(tmp.ul[L]);
165  tmp.uq = vq;
166  v[1] = (digit)HHALF(tmp.ul[H]);
167  v[2] = (digit)LHALF(tmp.ul[H]);
168  v[3] = (digit)HHALF(tmp.ul[L]);
169  v[4] = (digit)LHALF(tmp.ul[L]);
170  for (n = 4; v[1] == 0; v++) {
171  if (--n == 1) {
172  uint32_t rbj; /* r*B+u[j] (not root boy jim) */
173  digit q1, q2, q3, q4;
174 
175  /*
176  * Change of plan, per exercise 16.
177  * r = 0;
178  * for j = 1..4:
179  * q[j] = floor((r*B + u[j]) / v),
180  * r = (r*B + u[j]) % v;
181  * We unroll this completely here.
182  */
183  t = v[2]; /* nonzero, by definition */
184  q1 = (digit)(u[1] / t);
185  rbj = COMBINE(u[1] % t, u[2]);
186  q2 = (digit)(rbj / t);
187  rbj = COMBINE(rbj % t, u[3]);
188  q3 = (digit)(rbj / t);
189  rbj = COMBINE(rbj % t, u[4]);
190  q4 = (digit)(rbj / t);
191  if (arq)
192  *arq = rbj % t;
193  tmp.ul[H] = COMBINE(q1, q2);
194  tmp.ul[L] = COMBINE(q3, q4);
195  return (tmp.q);
196  }
197  }
198 
199  /*
200  * By adjusting q once we determine m, we can guarantee that
201  * there is a complete four-digit quotient at &qspace[1] when
202  * we finally stop.
203  */
204  for (m = 4 - n; u[1] == 0; u++)
205  m--;
206  for (i = 4 - m; --i >= 0;)
207  q[i] = 0;
208  q += 4 - m;
209 
210  /*
211  * Here we run Program D, translated from MIX to C and acquiring
212  * a few minor changes.
213  *
214  * D1: choose multiplier 1 << d to ensure v[1] >= B/2.
215  */
216  d = 0;
217  for (t = v[1]; t < B / 2; t <<= 1)
218  d++;
219  if (d > 0) {
220  shl(&u[0], m + n, d); /* u <<= d */
221  shl(&v[1], n - 1, d); /* v <<= d */
222  }
223  /*
224  * D2: j = 0.
225  */
226  j = 0;
227  v1 = v[1]; /* for D3 -- note that v[1..n] are constant */
228  v2 = v[2]; /* for D3 */
229  do {
230  digit uj0, uj1, uj2;
231 
232  /*
233  * D3: Calculate qhat (\^q, in TeX notation).
234  * Let qhat = min((u[j]*B + u[j+1])/v[1], B-1), and
235  * let rhat = (u[j]*B + u[j+1]) mod v[1].
236  * While rhat < B and v[2]*qhat > rhat*B+u[j+2],
237  * decrement qhat and increase rhat correspondingly.
238  * Note that if rhat >= B, v[2]*qhat < rhat*B.
239  */
240  uj0 = u[j + 0]; /* for D3 only -- note that u[j+...] change */
241  uj1 = u[j + 1]; /* for D3 only */
242  uj2 = u[j + 2]; /* for D3 only */
243  if (uj0 == v1) {
244  qhat = B;
245  rhat = uj1;
246  goto qhat_too_big;
247  } else {
248  uint32_t nn = COMBINE(uj0, uj1);
249  qhat = nn / v1;
250  rhat = nn % v1;
251  }
252  while (v2 * qhat > COMBINE(rhat, uj2)) {
253  qhat_too_big:
254  qhat--;
255  if ((rhat += v1) >= B)
256  break;
257  }
258  /*
259  * D4: Multiply and subtract.
260  * The variable `t' holds any borrows across the loop.
261  * We split this up so that we do not require v[0] = 0,
262  * and to eliminate a final special case.
263  */
264  for (t = 0, i = n; i > 0; i--) {
265  t = u[i + j] - v[i] * qhat - t;
266  u[i + j] = (digit)LHALF(t);
267  t = (B - HHALF(t)) & (B - 1);
268  }
269  t = u[j] - t;
270  u[j] = (digit)LHALF(t);
271  /*
272  * D5: test remainder.
273  * There is a borrow if and only if HHALF(t) is nonzero;
274  * in that (rare) case, qhat was too large (by exactly 1).
275  * Fix it by adding v[1..n] to u[j..j+n].
276  */
277  if (HHALF(t)) {
278  qhat--;
279  for (t = 0, i = n; i > 0; i--) { /* D6: add back. */
280  t += u[i + j] + v[i];
281  u[i + j] = (digit)LHALF(t);
282  t = HHALF(t);
283  }
284  u[j] = (digit)LHALF(u[j] + t);
285  }
286  q[j] = (digit)qhat;
287  }while (++j <= m); /* D7: loop on j. */
288 
289  /*
290  * If caller wants the remainder, we have to calculate it as
291  * u[m..m+n] >> d (this is at most n digits and thus fits in
292  * u[m+1..m+n], but we may need more source digits).
293  */
294  if (arq) {
295  if (d) {
296  for (i = m + n; i > m; --i)
297  u[i] = (digit)(((uint32_t)u[i] >> d) |
298  LHALF((uint32_t)u[i - 1] << (HALF_BITS - d)));
299  u[i] = 0;
300  }
301  tmp.ul[H] = COMBINE(uspace[1], uspace[2]);
302  tmp.ul[L] = COMBINE(uspace[3], uspace[4]);
303  *arq = tmp.q;
304  }
305 
306  tmp.ul[H] = COMBINE(qspace[1], qspace[2]);
307  tmp.ul[L] = COMBINE(qspace[3], qspace[4]);
308  return (tmp.q);
309 }

311 /*
312  * Shift p[0]..p[len] left `sh' bits, ignoring any bits that
313  * `fall out' the left (there never will be any such anyway).
314  * We may assume len >= 0. NOTE THAT THIS WRITES len+1 DIGITS.
315  */
316 static void
317 shl(digit *p, int len, int sh)
318 {
319  int i;
320 
321  for (i = 0; i < len; i++)
322  p[i] = (digit)(LHALF((uint32_t)p[i] << sh) |
323  ((uint32_t)p[i + 1] >> (HALF_BITS - sh)));
324  p[i] = (digit)(LHALF((uint32_t)p[i] << sh));
325 }
