99 static void walkFdtNodes(const fdt_node_t* pkNode)
100 {
101  const char* pCompatible;
102  driver_t* pDriver;
103  uintptr_t driverTableCursor;
104  size_t propLen;
105  OS_RETURN_E retCode;
106 
107  if(pkNode == NULL)
108  {
109  return;
110  }
111 
112  /* Get the node compatible */
113  pCompatible = fdtGetProp(pkNode, COMPATIBLE_PROP_NAME, &propLen);
114  if(pCompatible != NULL && propLen > 0)
115  {
116  KERNEL_DEBUG(DEVMGR_DEBUG_ENABLED,
117  MODULE_NAME,
118  "Detected %s",
119  pCompatible);
120 
121  /* Get the head of the registered drivers section */
122  driverTableCursor = (uintptr_t)&_START_DRV_TABLE_ADDR;
123  pDriver = *(driver_t**)driverTableCursor;
124 
125  /* Compare with the list of registered drivers */
126  while(pDriver != NULL)
127  {
128  if(strcmp(pDriver->pCompatible, pCompatible) == 0)
129  {
130  retCode = pDriver->pDriverAttach(pkNode);
131  if(retCode != OS_NO_ERR)
132  {
133  KERNEL_ERROR("Failed to attach driver %s. Error %d\n",
134  pDriver->pName,
135  retCode);
136  }
137  else
138  {
139  KERNEL_DEBUG(DEVMGR_DEBUG_ENABLED,
140  MODULE_NAME,
141  "%s attached successfully.",
142  pDriver->pName,
143  pCompatible);
144  }
145 
146  break;
147  }
148  else
149  {
150  KERNEL_DEBUG(DEVMGR_DEBUG_ENABLED,
151  MODULE_NAME,
152  "%s not compatible with %s.",
153  pDriver->pCompatible,
154  pCompatible);
155  }
156  driverTableCursor += sizeof(uintptr_t);
157  pDriver = *(driver_t**)driverTableCursor;
158  }
159 
160  }
161 
162  /* Got to next nodes */
163  walkFdtNodes(fdtGetChild(pkNode));
164  walkFdtNodes(fdtGetNextNode(pkNode));
165 }
