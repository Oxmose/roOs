153 OS_RETURN_E kernel_exception_register_handler(const uint32_t exception_line,
154  custom_handler_t handler)
155 {
156  uint32_t int_state;
157 
158 #ifdef ARCH_64_BITS
159  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REGISTER_START, 3,
160  exception_line,
161  (uintptr_t)handler & 0xFFFFFFFF,
162  (uintptr_t)handler >> 32);
163 #else
164  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REGISTER_START, 3,
165  exception_line,
166  (uintptr_t)handler & 0xFFFFFFFF,
167  (uintptr_t)0);
168 #endif
169 
170  if((int32_t)exception_line < MIN_EXCEPTION_LINE ||
171  exception_line > MAX_EXCEPTION_LINE)
172  {
173  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REGISTER_END, 2,
174  exception_line,
175  OR_ERR_UNAUTHORIZED_INTERRUPT_LINE);
176 
177  return OR_ERR_UNAUTHORIZED_INTERRUPT_LINE;
178  }
179 
180  if(handler == NULL)
181  {
182  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REGISTER_END, 2,
183  exception_line,
184  OS_ERR_NULL_POINTER);
185 
186  return OS_ERR_NULL_POINTER;
187  }
188 
189  ENTER_CRITICAL(int_state);
190 
191  if(kernel_interrupt_handlers[exception_line] != NULL)
192  {
193  EXIT_CRITICAL(int_state);
194  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REGISTER_END, 2,
195  exception_line,
196  OS_ERR_INTERRUPT_ALREADY_REGISTERED);
197 
198  return OS_ERR_INTERRUPT_ALREADY_REGISTERED;
199  }
200 
201  kernel_interrupt_handlers[exception_line] = handler;
202 
203  KERNEL_DEBUG(EXCEPTIONS_DEBUG_ENABLED, "EXCEPTIONS",
204  "Added exception %u handler at 0x%p", exception_line, handler);
205 
206  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REGISTER_END, 2,
207  exception_line,
208  OS_NO_ERR);
209 
210  EXIT_CRITICAL(int_state);
211  return OS_NO_ERR;
212 }

218  KERNEL_TRACE_EVENT(EVENT_KERNEL_EXCEPTION_REMOVE_START, 1,

230 
231  ENTER_CRITICAL(int_state);
232

234  {
235  EXIT_CRITICAL(int_state);

237  exception_line,

244

249  exception_line,
