139 inline static uint32_t _cpu_get_cpuid_max (const uint32_t ext)
140 {
141  uint32_t regs[4];
142 
143  /* Host supports CPUID. Return highest supported CPUID input value. */
144  __asm__ __volatile__("cpuid":"=a"(*regs),"=b"(*(regs+1)),

162 inline static int32_t _cpu_cpuid(const uint32_t code, uint32_t regs[4])
163 {
164  uint32_t ext = code & 0x80000000;
165  uint32_t maxlevel = _cpu_get_cpuid_max(ext);
166 
167  if (maxlevel == 0 || maxlevel < code)
168  {
169  return 0;
170  }
171  __asm__ __volatile__("cpuid":"=a"(*regs),"=b"(*(regs+1)),
172  "=c"(*(regs+2)),"=d"(*(regs+3)):"a"(code));
173  return 1;
174 }
