167 void kernel_interrupt_handler(void)
168 {
169  custom_handler_t handler;
170  kernel_thread_t* current_thread;
171  uint32_t int_id;
172 
173 
174  /* Get the current thread */
175  current_thread = scheduler_get_current_thread();
176  int_id = current_thread->v_cpu.int_context.int_id;
177 
178  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_HANDLER_START, 2,
179  int_id, current_thread->tid);
180 
181  /* If interrupts are disabled */
182  if(_cpu_get_saved_interrupt_state(&current_thread->v_cpu) == 0 &&
183  int_id != PANIC_INT_LINE &&
184  int_id != SCHEDULER_SW_INT_LINE &&
185  int_id >= MIN_INTERRUPT_LINE)
186  {
187  KERNEL_DEBUG(INTERRUPTS_DEBUG_ENABLED, MODULE_NAME,
188  "Blocked interrupt %u", int_id);
189  return;
190  }
191 
192  if(int_id == PANIC_INT_LINE)
193  {
194  panic_handler(current_thread);
195  }
196 
197  KERNEL_DEBUG(INTERRUPTS_DEBUG_ENABLED, MODULE_NAME, "Int %d", int_id);
198 
199  /* Check for spurious interrupt */
200  if(interrupt_driver.driver_handle_spurious(int_id) ==
201  INTERRUPT_TYPE_SPURIOUS)
202  {
203  _spurious_handler();
204  return;
205  }
206 
207  KERNEL_DEBUG(INTERRUPTS_DEBUG_ENABLED, MODULE_NAME, "Non spurious %d",
208  int_id);
209 
210  /* Select custom handlers */
211  if(int_id < INT_ENTRY_COUNT &&
212  kernel_interrupt_handlers[int_id] != NULL)
213  {
214  handler = kernel_interrupt_handlers[int_id];
215  }
216  else
217  {
218  handler = panic_handler;
219  }
220 
221  /* Execute the handler */
222  handler(current_thread);
223 
224  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_HANDLER_END, 2,
225  int_id, current_thread->tid);
226 }

228 void kernel_interrupt_init(void)
229 {
230  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_INIT_START, 0);
231  KERNEL_DEBUG(INTERRUPTS_DEBUG_ENABLED, MODULE_NAME,
232  "Initializing interrupt manager.");
233 
234  /* Blank custom interrupt handlers */
235  memset(kernel_interrupt_handlers,
236  0,
237  sizeof(custom_handler_t) * INT_ENTRY_COUNT);
238 
239  /* Attach the special PANIC interrupt for when we don't know what to do */
240  kernel_interrupt_handlers[PANIC_INT_LINE] = panic_handler;
241 
242  /* Init state */
243  kernel_interrupt_disable();
244  spurious_interrupt_count = 0;
245 
246  /* Init driver */
247  interrupt_driver.driver_get_irq_int_line = _init_driver_get_irq_int_line;
248  interrupt_driver.driver_handle_spurious = _init_driver_handle_spurious;
249  interrupt_driver.driver_set_irq_eoi = _init_driver_set_irq_eoi;
250  interrupt_driver.driver_set_irq_mask = _init_driver_set_irq_mask;
251 
252  TEST_POINT_FUNCTION_CALL(interrupt_test, TEST_INTERRUPT_ENABLED);
253 
254  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_INIT_END, 0);
255 }

298 OS_RETURN_E kernel_interrupt_register_int_handler(const uint32_t interrupt_line,
299  custom_handler_t handler)
300 {
301  uint32_t int_state;
302 
303 #ifdef ARCH_64_BITS
304  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_REGISTER_START, 3,
305  interrupt_line,
306  (uintptr_t)handler & 0xFFFFFFFF,
307  (uintptr_t)handler >> 32);
308 #else
309  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_REGISTER_START, 3,
310  interrupt_line,
311  (uintptr_t)handler & 0xFFFFFFFF,
312  (uintptr_t)0);
313 #endif
314 
315  if(interrupt_line < MIN_INTERRUPT_LINE ||
316  interrupt_line > MAX_INTERRUPT_LINE)
317  {
318  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_REGISTER_END, 2,
319  interrupt_line,
320  OR_ERR_UNAUTHORIZED_INTERRUPT_LINE);
321 
322  return OR_ERR_UNAUTHORIZED_INTERRUPT_LINE;
323  }
324 
325  if(handler == NULL)
326  {
327  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_REGISTER_END, 2,
328  interrupt_line,
329  OS_ERR_NULL_POINTER);
330 
331  return OS_ERR_NULL_POINTER;
332  }
333 
334  ENTER_CRITICAL(int_state);
335 
336  if(kernel_interrupt_handlers[interrupt_line] != NULL)
337  {
338  EXIT_CRITICAL(int_state);
339  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_REGISTER_END, 2,
340  interrupt_line,
341  OS_ERR_INTERRUPT_ALREADY_REGISTERED);
342 
343  return OS_ERR_INTERRUPT_ALREADY_REGISTERED;
344  }
345 
346  kernel_interrupt_handlers[interrupt_line] = handler;
347 
348  KERNEL_DEBUG(INTERRUPTS_DEBUG_ENABLED, MODULE_NAME,
349  "Added INT %u handler at 0x%p", interrupt_line, handler);
350 
351  KERNEL_TRACE_EVENT(EVENT_KERNEL_INTERRUPT_REGISTER_END, 2,
352  interrupt_line,
353  OS_NO_ERR);
354 
355  EXIT_CRITICAL(int_state);
356 
357  return OS_NO_ERR;
358 }
