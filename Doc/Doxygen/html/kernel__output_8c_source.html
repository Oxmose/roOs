255  while(*string != 0)

257  /* If the character is lowercase, makes it uppercase */

280 static void _formater(const char* str,
281  __builtin_va_list args,
282  output_t used_output)
283 {
284  size_t pos;
285  size_t str_length;
286  uint64_t seq_val;
287  size_t str_size;
288 
289 
290  uint8_t modifier;
291 
292  uint8_t length_mod;
293  uint8_t padding_mod;
294  bool_t upper_mod;
295  char pad_char_mod;
296 
297  char tmp_seq[128];
298 
299  char* args_value;
300 
301  modifier = 0;
302  length_mod = 4;
303  padding_mod = 0;
304  upper_mod = FALSE;
305  pad_char_mod = ' ';
306  str_length = strlen(str);
307 
308  for(pos = 0; pos < str_length; ++pos)
309  {
310  if(str[pos] == '%')
311  {
312  /* If we encouter this character in a modifier sequence, it was
313  * just an escape one.
314  */
315  modifier = !modifier;
316  if(modifier)
317  {
318  continue;
319  }
320  else
321  {
322  used_output.putc(str[pos]);
323  }
324  }
325  else if(modifier)
326  {
327  switch(str[pos])
328  {
329  /* Length mods */
330  case 'h':
331  length_mod /= 2;
332  continue;
333  case 'l':
334  length_mod *= 2;
335  continue;
336 
337  /* Specifier mods */
338  case 's':
339  args_value = __builtin_va_arg(args, char*);
340  used_output.puts(args_value);
341  break;
342  case 'd':
343  case 'i':
344  GET_SEQ_VAL(seq_val, args, length_mod);
345  memset(tmp_seq, 0, sizeof(tmp_seq));
346  itoa(seq_val, tmp_seq, 10);
347  PAD_SEQ
348  used_output.puts(tmp_seq);
349  break;
350  case 'u':
351  GET_SEQ_VAL(seq_val, args, length_mod);
352  memset(tmp_seq, 0, sizeof(tmp_seq));
353  uitoa(seq_val, tmp_seq, 10);
354  PAD_SEQ
355  used_output.puts(tmp_seq);
356  break;
357  case 'X':
358  upper_mod = TRUE;
359  __attribute__ ((fallthrough));
360  case 'x':
361  GET_SEQ_VAL(seq_val, args, length_mod);
362  memset(tmp_seq, 0, sizeof(tmp_seq));
363  uitoa(seq_val, tmp_seq, 16);
364  PAD_SEQ
365  if(upper_mod == TRUE)
366  {
367  _toupper(tmp_seq);
368  }
369  else
370  {
371  _tolower(tmp_seq);
372  }
373  used_output.puts(tmp_seq);
374  break;
375  case 'P':
376  upper_mod = TRUE;
377  __attribute__ ((fallthrough));
378  case 'p':
379  padding_mod = 2 * sizeof(uintptr_t);
380  pad_char_mod = '0';
381  length_mod = sizeof(uintptr_t);
382  GET_SEQ_VAL(seq_val, args, length_mod);
383  memset(tmp_seq, 0, sizeof(tmp_seq));
384  uitoa(seq_val, tmp_seq, 16);
385  PAD_SEQ
386  if(upper_mod == TRUE)
387  {
388  _toupper(tmp_seq);
389  }
390  else
391  {
392  _tolower(tmp_seq);
393  }
394  used_output.puts(tmp_seq);
395  break;
396  case 'c':
397  length_mod = sizeof(char);
398  GET_SEQ_VAL(tmp_seq[0], args, length_mod);
399  used_output.putc(tmp_seq[0]);
400  break;
401 
402  /* Padding mods */
403  case '0':
404  if(padding_mod == 0)
405  {
406  pad_char_mod = '0';
407  }
408  else
409  {
410  padding_mod *= 10;
411  }
412  continue;
413  case '1':
414  padding_mod = padding_mod * 10 + 1;
415  continue;
416  case '2':
417  padding_mod = padding_mod * 10 + 2;
418  continue;
419  case '3':
420  padding_mod = padding_mod * 10 + 3;
421  continue;
422  case '4':
423  padding_mod = padding_mod * 10 + 4;
424  continue;
425  case '5':
426  padding_mod = padding_mod * 10 + 5;
427  continue;
428  case '6':
429  padding_mod = padding_mod * 10 + 6;
430  continue;
431  case '7':
432  padding_mod = padding_mod * 10 + 7;
433  continue;
434  case '8':
435  padding_mod = padding_mod * 10 + 8;
436  continue;
437  case '9':
438  padding_mod = padding_mod * 10 + 9;
439  continue;
440  default:
441  continue;
442  }
443  }
444  else
445  {
446  used_output.putc(str[pos]);
447  }
448 
449  /* Reinit mods */
450  length_mod = 4;
451  padding_mod = 0;
452  upper_mod = FALSE;
453  pad_char_mod = ' ';
454  modifier = 0;
455 
456  }
457 }
