510 void* kmalloc(size_t size)
511 {
512  size_t n;
513  mem_chunk_t* chunk;
514  mem_chunk_t* chunk2;
515  size_t size2;
516  size_t len;
517  uint32_t int_state;
518 
519  KERNEL_DEBUG(KHEAP_DEBUG_ENABLED, "KHEAP",
520  "Kheap allocating %uB (%uB free, %uB used)",
521  size, mem_free,
522  kheap_init_free - mem_free);
523 
524  if(init == FALSE || size == 0)
525  {
526  return NULL;
527  }
528 
529  ENTER_CRITICAL(int_state);
530 
531  size = (size + ALIGN - 1) & (~(ALIGN - 1));
532 
533  if (size < MIN_SIZE)
534  {
535  size = MIN_SIZE;
536  }
537 
538  n = _memory_chunk_slot(size - 1) + 1;
539 
540  if (n >= NUM_SIZES)
541  {
542  EXIT_CRITICAL(int_state);
543  return NULL;
544  }
545 
546  while(free_chunk[n] == 0)
547  {
548  ++n;
549  if (n >= NUM_SIZES)
550  {
551  EXIT_CRITICAL(int_state);
552  return NULL;
553  }
554  }
555 
556  chunk = LIST_POP(&free_chunk[n], free);
557  size2 = _memory_chunk_size(chunk);
558  len = 0;
559 
560  if (size + sizeof(mem_chunk_t) <= size2)
561  {
562  chunk2 = (mem_chunk_t*)(((int8_t*)chunk) + HEADER_SIZE + size);
563 
564  _memory_chunk_init(chunk2);
565 
566  _insert_after(&chunk->all, &chunk2->all);
567 
568  len = _memory_chunk_size(chunk2);
569  n = _memory_chunk_slot(len);
570 
571  LIST_PUSH(&free_chunk[n], chunk2, free);
572 
573  mem_meta += HEADER_SIZE;
574  mem_free += len;
575  }
576 
577  chunk->used = TRUE;
578 
579  mem_free -= size2;
580 
581  KERNEL_DEBUG(KHEAP_DEBUG_ENABLED, "KHEAP",
582  "Kheap allocated 0x%p -> %uB (%uB free, %uB used)",
583  chunk->data, size2 - len - HEADER_SIZE, mem_free,
584  kheap_init_free - mem_free);
585 
586  EXIT_CRITICAL(int_state);
587 
588 #ifdef ARCH_64_BITS
589  KERNEL_TRACE_EVENT(EVENT_KERNEL_HEAP_ALLOC, 4,
590  (uintptr_t)chunk->data & 0xFFFFFFFF,
591  (uintptr_t)chunk->data >> 32,
592  size,
593  mem_free);
594 #else
595  KERNEL_TRACE_EVENT(EVENT_KERNEL_HEAP_ALLOC, 4,
596  (uintptr_t)chunk->data & 0xFFFFFFFF,
597  (uintptr_t)0,
598  size,
599  mem_free);
600 #endif
601 
602  return chunk->data;
603 }

606 {
607  uint32_t used;
608  mem_chunk_t* chunk;
609  mem_chunk_t* next;
610  mem_chunk_t* prev;
611  uint32_t int_state;
612 
613  if(init == FALSE || ptr == NULL)
614  {
615  return;
616  }
617 
618  ENTER_CRITICAL(int_state);
619 
620  chunk = (mem_chunk_t*)((int8_t*)ptr - HEADER_SIZE);
621  next = CONTAINER(mem_chunk_t, all, chunk->all.next);
622  prev = CONTAINER(mem_chunk_t, all, chunk->all.prev);
623 
624  used = _memory_chunk_size(chunk);
625 
626  if (next->used == FALSE)
627  {
628  _remove_free(next);
629  _remove(&next->all);
630 
631  mem_meta -= HEADER_SIZE;
632  mem_free += HEADER_SIZE;
633  }
634 
635  if (prev->used == FALSE)
636  {
637  _remove_free(prev);
638  _remove(&chunk->all);
639 
640  _push_free(prev);
641  mem_meta -= HEADER_SIZE;
642  mem_free += HEADER_SIZE;
643  }
644  else
645  {
646  chunk->used = FALSE;
647  LIST_INIT(chunk, free);
648  _push_free(chunk);
649  }

651  KERNEL_DEBUG(KHEAP_DEBUG_ENABLED, "KHEAP",
652  "[KHEAP] Kheap freed 0x%p -> %uB", ptr, used);

654 #ifdef ARCH_64_BITS
655  KERNEL_TRACE_EVENT(EVENT_KERNEL_HEAP_FREE, 3,
656  (uintptr_t)ptr & 0xFFFFFFFF,

662  (uintptr_t)0,

664 #endif
