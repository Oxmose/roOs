224 void pit_set_frequency(const uint32_t freq)
225 {
226  uint32_t int_state;
227 
228  KERNEL_TRACE_EVENT(EVENT_KERNEL_PIT_SET_FREQ_START, 1, freq);
229 
230  PIT_ASSERT((freq >= PIT_MIN_FREQ && freq <= PIT_MAX_FREQ),
231  "PIT timer frequency out of bound",
232  OS_ERR_INCORRECT_VALUE);
233 
234  ENTER_CRITICAL(int_state);
235 
236  /* Disable PIT IRQ */
237  pit_disable();
238 
239  tick_freq = freq;
240 
241  /* Set clock frequency */
242  uint16_t tick_freq = (uint16_t)((uint32_t)PIT_QUARTZ_FREQ / freq);
243  _cpu_outb(PIT_COMM_SET_FREQ, PIT_COMM_PORT);
244  _cpu_outb(tick_freq & 0x00FF, PIT_DATA_PORT);
245  _cpu_outb(tick_freq >> 8, PIT_DATA_PORT);
246 
247  KERNEL_DEBUG(PIT_DEBUG_ENABLED, MODULE_NAME, "New PIT frequency set (%d)", freq);
248 
249  EXIT_CRITICAL(int_state);
250 
251  KERNEL_TRACE_EVENT(EVENT_KERNEL_PIT_SET_FREQ_END, 1, freq);
252 
253  /* Enable PIT IRQ */
254  pit_enable();
255 }

262 OS_RETURN_E pit_set_handler(void(*handler)(kernel_thread_t*))
263 {
264  OS_RETURN_E err;
265  uint32_t int_state;
266 
267 #ifdef ARCH_64_BITS
268  KERNEL_TRACE_EVENT(EVENT_KERNEL_PIT_SET_HANDLER, 2,
269  (uintptr_t)handler & 0xFFFFFFFF,
270  (uintptr_t)handler >> 32);
271 #else
272  KERNEL_TRACE_EVENT(EVENT_KERNEL_PIT_SET_HANDLER, 2,
273  (uintptr_t)handler & 0xFFFFFFFF,
274  (uintptr_t)0);
275 #endif
276 
277  if(handler == NULL)
278  {
279  return OS_ERR_NULL_POINTER;
280  }
281 
282  ENTER_CRITICAL(int_state);
283 
284  pit_disable();
285 
286  /* Remove the current handler */
287  err = kernel_interrupt_remove_irq_handler(PIT_IRQ_LINE);
288  if(err != OS_NO_ERR)
289  {
290  EXIT_CRITICAL(int_state);
291  pit_enable();
292  return err;
293  }
294 
295  err = kernel_interrupt_register_irq_handler(PIT_IRQ_LINE, handler);
296  if(err != OS_NO_ERR)
297  {
298  EXIT_CRITICAL(int_state);
299  pit_enable();
300  return err;
301  }
302 
303  KERNEL_DEBUG(PIT_DEBUG_ENABLED, MODULE_NAME, "New PIT handler set 0x%p", handler);
304 
305  EXIT_CRITICAL(int_state);
306  pit_enable();
307 
308  return err;
309 }
