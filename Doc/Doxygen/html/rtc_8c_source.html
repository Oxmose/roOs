189 void rtc_init(void)
190 {
191  OS_RETURN_E err;
192  int8_t prev_ored;
193  int8_t prev_rate;
194 
195  KERNEL_TRACE_EVENT(EVENT_KERNEL_RTC_INIT_START, 0);
196 
197  /* Init system times */
198  disabled_nesting = 1;
199 
200  /* Init real times */
201  day_time = 0;
202  date.weekday = 0;
203  date.day = 0;
204  date.month = 0;
205  date.year = 0;
206 
207  /* Init CMOS IRQ8 */
208  _cpu_outb((CMOS_NMI_DISABLE_BIT << 7) | CMOS_REG_B, CMOS_COMM_PORT);
209  prev_ored = _cpu_inb(CMOS_DATA_PORT);
210  _cpu_outb((CMOS_NMI_DISABLE_BIT << 7) | CMOS_REG_B, CMOS_COMM_PORT);
211  _cpu_outb(prev_ored | CMOS_ENABLE_RTC, CMOS_DATA_PORT);
212 
213  /* Init CMOS IRQ8 rate */
214  _cpu_outb((CMOS_NMI_DISABLE_BIT << 7) | CMOS_REG_A, CMOS_COMM_PORT);
215  prev_rate = _cpu_inb(CMOS_DATA_PORT);
216  _cpu_outb((CMOS_NMI_DISABLE_BIT << 7) | CMOS_REG_A, CMOS_COMM_PORT);
217  _cpu_outb((prev_rate & 0xF0) | RTC_INIT_RATE, CMOS_DATA_PORT);
218  rtc_frequency = (RTC_QUARTZ_FREQ >> (RTC_INIT_RATE - 1));
219 
220  /* Set rtc clock interrupt handler */
221  err = kernel_interrupt_register_irq_handler(RTC_IRQ_LINE, _dummy_handler);
222  RTC_ASSERT(err == OS_NO_ERR, "Could not register RTC handler", err);
223 
224  /* Clear mask before setting IRQ */
225  kernel_interrupt_set_irq_mask(RTC_IRQ_LINE, 0);
226 
227  /* Just dummy read register C to unlock interrupt */
228  _cpu_outb(CMOS_REG_C, CMOS_COMM_PORT);
229  _cpu_inb(CMOS_DATA_PORT);
230 
231  time_register_rtc_manager(rtc_update_time);
232 
233  KERNEL_DEBUG(RTC_DEBUG_ENABLED, MODULE_NAME, "RTC Initialized");
234 
235  KERNEL_TRACE_EVENT(EVENT_KERNEL_RTC_INIT_END, 1, KERNEL_RTC_TIMER_FREQ);
236 }

287 void rtc_set_frequency(const uint32_t frequency)
288 {
289  uint32_t prev_rate;
290  uint32_t rate;
291  uint32_t int_state;
292 
293  KERNEL_TRACE_EVENT(EVENT_KERNEL_RTC_SET_FREQ_START, 1, frequency);
294 
295  if(frequency < RTC_MIN_FREQ || frequency > RTC_MAX_FREQ)
296  {
297  rtc_frequency = 0;
298  KERNEL_ERROR("RTC timer frequency out of bound %u not in [%u:%u]\n",
299  frequency,
300  RTC_MIN_FREQ,
301  RTC_MAX_FREQ);
302  return;
303  }
304 
305  /* Choose the closest rate to the frequency */
306  if(frequency < 4)
307  {
308  rate = 15;
309  }
310  else if(frequency < 8)
311  {
312  rate = 14;
313  }
314  else if(frequency < 16)
315  {
316  rate = 13;
317  }
318  else if(frequency < 32)
319  {
320  rate = 12;
321  }
322  else if(frequency < 64)
323  {
324  rate = 11;
325  }
326  else if(frequency < 128)
327  {
328  rate = 10;
329  }
330  else if(frequency < 256)
331  {
332  rate = 9;
333  }
334  else if(frequency < 512)
335  {
336  rate = 8;
337  }
338  else if(frequency < 1024)
339  {
340  rate = 7;
341  }
342  else if(frequency < 2048)
343  {
344  rate = 6;
345  }
346  else if(frequency < 4096)
347  {
348  rate = 5;
349  }
350  else if(frequency < 8192)
351  {
352  rate = 4;
353  }
354  else
355  {
356  rate = 3;
357  }
358 
359  ENTER_CRITICAL(int_state);
360 
361  /* Disable RTC IRQ */
362  rtc_disable();
363 
364  /* Set clock frequency */
365  /* Init CMOS IRQ8 rate */
366  _cpu_outb((CMOS_NMI_DISABLE_BIT << 7) | CMOS_REG_A, CMOS_COMM_PORT);
367  prev_rate = _cpu_inb(CMOS_DATA_PORT);
368  _cpu_outb((CMOS_NMI_DISABLE_BIT << 7) | CMOS_REG_A, CMOS_COMM_PORT);
369  _cpu_outb((prev_rate & 0xF0) | rate, CMOS_DATA_PORT);
370 
371  rtc_frequency = (RTC_QUARTZ_FREQ >> (rate - 1));
372 
373  KERNEL_DEBUG(RTC_DEBUG_ENABLED, MODULE_NAME, "New RTC rate set (%d: %dHz)",
374  rate,
375  rtc_frequency);
376 
377  EXIT_CRITICAL(int_state);
378 
379  KERNEL_TRACE_EVENT(EVENT_KERNEL_RTC_SET_FREQ_END, 1, frequency);
380 
381  /* Enable RTC IRQ */
382  rtc_enable();
383 }

478 void rtc_update_time(void)
479 {
480  uint8_t seconds;
481  uint8_t minutes;
482  uint32_t hours;
483  uint8_t century;
484  uint8_t reg_b;
485 
486  KERNEL_TRACE_EVENT(EVENT_KERNEL_RTC_UPDATE_TIME_START, 0);
487 
488  /* Set time */
489  /* Select CMOS seconds register and read */
490  _cpu_outb(CMOS_SECONDS_REGISTER, CMOS_COMM_PORT);
491  seconds = _cpu_inb(CMOS_DATA_PORT);
492 
493  /* Select CMOS minutes register and read */
494  _cpu_outb(CMOS_MINUTES_REGISTER, CMOS_COMM_PORT);
495  minutes = _cpu_inb(CMOS_DATA_PORT);
496 
497  /* Select CMOS hours register and read */
498  _cpu_outb(CMOS_HOURS_REGISTER, CMOS_COMM_PORT);
499  hours = _cpu_inb(CMOS_DATA_PORT);
500 
501  /* Select CMOS day register and read */
502  _cpu_outb(CMOS_DAY_REGISTER, CMOS_COMM_PORT);
503  date.day = _cpu_inb(CMOS_DATA_PORT);
504 
505  /* Select CMOS month register and read */
506  _cpu_outb(CMOS_MONTH_REGISTER, CMOS_COMM_PORT);
507  date.month = _cpu_inb(CMOS_DATA_PORT);
508 
509  /* Select CMOS years register and read */
510  _cpu_outb(CMOS_YEAR_REGISTER, CMOS_COMM_PORT);
511  date.year = _cpu_inb(CMOS_DATA_PORT);
512 
513  /* Select CMOS century register and read */
514  if(CMOS_CENTURY_REGISTER != 0)
515  {
516  _cpu_outb(CMOS_CENTURY_REGISTER, CMOS_COMM_PORT);
517  century = _cpu_inb(CMOS_DATA_PORT);
518  }
519  else
520  {
521  century = CURRENT_YEAR / 100;
522  }
523 
524  /* Convert BCD to binary if necessary */
525  _cpu_outb(CMOS_REG_B, CMOS_COMM_PORT);
526  reg_b = _cpu_inb(CMOS_DATA_PORT);
527 
528  if((reg_b & 0x04) == 0)
529  {
530  seconds = (seconds & 0x0F) + ((seconds / 16) * 10);
531  minutes = (minutes & 0x0F) + ((minutes / 16) * 10);
532  hours = ((hours & 0x0F) + (((hours & 0x70) / 16) * 10)) |
533  (hours & 0x80);
534  date.day = (date.day & 0x0F) + ((date.day / 16) * 10);
535  date.month = (date.month & 0x0F) + ((date.month / 16) * 10);
536  date.year = (date.year & 0x0F) + ((date.year / 16) * 10);
537 
538  if(CMOS_CENTURY_REGISTER != 0)
539  {
540  century = (century & 0x0F) + ((century / 16) * 10);
541  }
542  }
543 
544  /* Convert to 24H */
545  if((reg_b & 0x02) == 0 && (hours & 0x80) >= 1)
546  {
547  hours = ((hours & 0x7F) + 12) % 24;
548  }
549 
550  /* Get year */
551  if(CMOS_CENTURY_REGISTER != 0)
552  {
553  date.year += century * 100;
554  }
555  else
556  {
557  date.year = date.year + 2000;
558  }
559 
560  /* Compute week day and day time */
561  date.weekday = ((date.day + date.month + date.year + date.year / 4)
562  + 1) % 7 + 1;
563  day_time = seconds + 60 * minutes + 3600 * hours;
564 
565  /* Clear C Register */
566  _cpu_outb(CMOS_REG_C, CMOS_COMM_PORT);
567  _cpu_inb(CMOS_DATA_PORT);
568 
569  KERNEL_TRACE_EVENT(EVENT_KERNEL_RTC_UPDATE_TIME_END, 0);
570 
571  KERNEL_DEBUG(RTC_DEBUG_ENABLED, MODULE_NAME, "Updated RTC");
572 }
