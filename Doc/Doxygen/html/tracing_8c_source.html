134  __builtin_va_start(args, field_count);

136  /* Check the cursor position and cycle if needed */
137  if(cursor + sizeof(event) > trace_buffer_size)
138  {
139  cursor = TRACE_LIB_HEADER_LEN;
140  }
141 
142  /* Write the event */
143  trace_buffer[cursor++] = (uint32_t)event;
144 
145  /* Write the timestamp */
146  timestpamp = time_get_current_uptime();
147  trace_buffer[cursor++] = (uint32_t)timestpamp;
148  trace_buffer[cursor++] = (uint32_t)(timestpamp >> 32);
149 
150  /* Write all metadata */
151  for(i = 0; i < field_count; ++i)
152  {
