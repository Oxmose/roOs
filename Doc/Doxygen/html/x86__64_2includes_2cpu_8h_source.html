156 inline static uint32_t _cpu_get_cpuid_max (const uint32_t ext)
157 {
158  uint32_t regs[4];
159 
160  /* Host supports CPUID. Return highest supported CPUID input value. */
161  __asm__ __volatile__("cpuid":"=a"(*regs),"=b"(*(regs+1)),
162  "=c"(*(regs+2)),"=d"(*(regs+3)):"a"(ext));
163 
164  return regs[0];
165 }

179 inline static int32_t _cpu_cpuid(const uint32_t code, uint32_t regs[4])
180 {
181  uint32_t ext = code & 0x80000000;
182  uint32_t maxlevel = _cpu_get_cpuid_max(ext);
183 
184  if (maxlevel == 0 || maxlevel < code)
185  {
186  return 0;
187  }
188  __asm__ __volatile__("cpuid":"=a"(*regs),"=b"(*(regs+1)),
189  "=c"(*(regs+2)),"=d"(*(regs+3)):"a"(code));
190  return 1;
191 }
